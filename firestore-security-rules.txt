// Firestore Security Rules for Export Access
// Copy and paste these rules into your Firebase Console > Firestore > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read access to all collections for export purposes
    // This allows anyone to read all documents (use with caution)
    
    match /{document=**} {
      allow read: if true;
      allow write: if false; // Prevent writes for security
    }
    
    // Alternative: More restrictive rules (recommended for production)
    // Uncomment these and comment out the above if you want more control
    
    /*
    // Ideas collection - allow read access
    match /ideas/{ideaId} {
      allow read: if true;
      allow write: if request.auth != null; // Only authenticated users can write
    }
    
    // Projects collection - allow read access
    match /projects/{projectId} {
      allow read: if true;
      allow write: if request.auth != null; // Only authenticated users can write
    }
    
    // Users collection - allow read access
    match /users/{userId} {
      allow read: if true;
      allow write: if request.auth != null; // Only authenticated users can write
    }
    */
  }
}

// IMPORTANT NOTES:
// 1. These rules allow READ access to all collections
// 2. WRITE access is disabled for security
// 3. After export, consider making rules more restrictive
// 4. For production, use the commented alternative rules
// 5. Always test rules in Firebase Console before deploying
